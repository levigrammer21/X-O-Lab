Connect Supabase

Enter your Supabase project URL and anon key. Data is stored in your browser until connected.

① Create a free project at supabase.com
② Go to Settings → API → copy Project URL + anon key
③ Run the SQL below in Supabase SQL Editor to create tables

Show setup SQL ▾ 
create table formations (
  id uuid primary key default gen_random_uuid(),
  name text not null,
  type text,
  notes text,
  created_at timestamptz default now()
);

create table plays (
  id uuid primary key default gen_random_uuid(),
  formation_id uuid references formations(id) on delete cascade,
  name text not null,
  notes text,
  data jsonb,
  created_at timestamptz default now()
);

-- Enable RLS + read-only public access
alter table formations enable row level security;
alter table plays enable row level security;
create policy "public read" on formations for select using (true);
create policy "public read" on plays for select using (true);
create policy "anon write" on formations for all using (true);
create policy "anon write" on plays for all using (true);
